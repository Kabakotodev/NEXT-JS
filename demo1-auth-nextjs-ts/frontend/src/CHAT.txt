import jakarta.persistence.*;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.Size;
import kabakoto.PkgAuth.entity.User;

@Entity
@Table(name = "vols")
public class Vol {
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable=true, length=30)
    private String codeVol = "NON RENSEIGNE";

    @Column(columnDefinition="TEXT")
    private String descVol;

    @Column(nullable=true, length=120)
    private String provenanceDestination = "NON RENSEIGNE";

    @Column(nullable=true, length=20)
    private String zone = "NON RENSEIGNE";

    @ManyToOne
    @JoinColumn(name = "compagnieId", nullable = false)
    private Compagnie compagnie;
