// Dans schema.prisma
model Document {
  id            Int      @id @default(autoincrement())
  nomDocument   String   @unique
  sigleDocument String   @unique
  descDocument  String

  regimes       Regime[] @relation("RegimeDocuments")  // ✅ AJOUT
  relations     CategorieRelation[]
}

model Categorie {
  id              Int      @id @default(autoincrement())
  nomCategorie    String
  descCategorie   String
  objetCategorie  String   @db.LongText

  regimes         Regime[] @relation("RegimeCategories") // ✅ AJOUT
  relations       CategorieRelation[]
}

model Nationalite {
  id            Int      @id @default(autoincrement())
  nomPays       String   @unique
  codePays      String   @unique
  nationaliteFr String
  nationaliteEn String

  regimes       Regime[] @relation("RegimeNationalites") // ✅ AJOUT
  relations     CategorieRelation[]
}

model CategorieRelation {
  id            Int @id @default(autoincrement())

  categorieId   Int
  documentId    Int
  nationaliteId Int

  categorie     Categorie   @relation(fields: [categorieId], references: [id])
  document      Document    @relation(fields: [documentId], references: [id])
  nationalite   Nationalite @relation(fields: [nationaliteId], references: [id])

  @@unique([documentId, nationaliteId])
}

interface SearchResult {
  visaRequired: boolean;
  visaType?: string;
  maxStayDays?: number;
  additionalDocuments?: string[];
  remarks?: string;
}

export const VisaSearchForm = () => {
  const [selectedCountry, setSelectedCountry] = useState<string>("");
  const [selectedDocument, setSelectedDocument] = useState<string>("");
  const [isLoading, setIsLoading] = useState(false);
  const [result, setResult] = useState<SearchResult | null>(null);
  const [hasSearched, setHasSearched] = useState(false);

  const handleSearch = async () => {
    if (!selectedCountry || !selectedDocument) return;

    setIsLoading(true);
    setHasSearched(true);

    // Simulation d'appel API - à remplacer par un vrai appel backend
    await new Promise((resolve) => setTimeout(resolve, 1200));

    // Résultats simulés selon le pays (pour démonstration)
    const mockResults: Record<string, SearchResult> = {
      fr: {
        visaRequired: false,
        maxStayDays: 90,
        remarks: "Les ressortissants français sont exemptés de visa pour les séjours touristiques de moins de 90 jours.",
      },
      us: {
        visaRequired: true,
        visaType: "Visa de court séjour",
        maxStayDays: 30,
        additionalDocuments: ["Attestation d'hébergement", "Justificatif de ressources financières", "Billet retour"],
        remarks: "Le visa doit être obtenu auprès de l'ambassade du Sénégal avant le départ.",
      },
      ci: {
        visaRequired: false,
        maxStayDays: 90,
        remarks: "Libre circulation dans l'espace CEDEAO pour les ressortissants ivoiriens.",
      },
    };

    setResult(mockResults[selectedCountry] || {
      visaRequired: true,
      visaType: "Visa de court séjour",
      maxStayDays: 30,
      additionalDocuments: ["Passeport valide 6 mois", "Photo d'identité", "Formulaire de demande"],
      remarks: "Veuillez contacter l'ambassade du Sénégal dans votre pays pour plus d'informations.",
    });

    setIsLoading(false);
  };

  const resetSearch = () => {
    setSelectedCountry("");
    setSelectedDocument("");
    setResult(null);
    setHasSearched(false);
  };

  return (
    <section id="recherche" className="py-16 lg:py-24 bg-muted/50">
      <div className="container mx-auto px-4">
        <div className="max-w-4xl mx-auto">
          {/* Section Header */}
          <div className="text-center mb-10">
            <h2 className="font-display text-3xl lg:text-4xl font-bold text-foreground mb-4">
              Vérifiez vos conditions d'accès
            </h2>
            <p className="text-lg text-muted-foreground max-w-2xl mx-auto">

            </p>
          </div>

          {/* Search Form Card */}
          <div className="card-elevated p-6 lg:p-8">

                <Select value={selectedCountry} onValueChange={setSelectedCountry}>
                  <SelectTrigger className="h-12 text-base">
                    <SelectValue placeholder="Sélectionnez votre pays" />
                  </SelectTrigger>
                  <SelectContent>
                    {countries.map((country) => (
                      <SelectItem key={country.id} value={country.id}>
                        {country.name}
                      </SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>

              {/* Document Type Select */}
              <div className="space-y-2">
                <label className="text-sm font-medium text-foreground flex items-center gap-2">
                  <FileText className="w-4 h-4 text-primary" />
                  Type de document de voyage
                </label>
                <Select value={selectedDocument} onValueChange={setSelectedDocument}>
                  <SelectTrigger className="h-12 text-base">
                    <SelectValue placeholder="Sélectionnez le type de document" />
                  </SelectTrigger>
                  <SelectContent>
                    {travelDocuments.map((doc) => (
                      <SelectItem key={doc.id} value={doc.id}>
                        {doc.label}
                      </SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>
            </div>

            {/* Search Button */}
            <div className="flex flex-col sm:flex-row gap-3">
              <Button
                variant="hero"
                size="lg"
                className="flex-1"
                onClick={handleSearch}
                disabled={!selectedCountry || !selectedDocument || isLoading}
              >
                {isLoading ? (
                  <>
                    <Loader2 className="w-5 h-5 animate-spin" />
                    Recherche en cours...
                  </>
                ) : (
                  <>
                    <Search className="w-5 h-5" />
                    Rechercher
                  </>
                )}
              </Button>
              {hasSearched && (
                <Button variant="outline" size="lg" onClick={resetSearch}>
                  Nouvelle recherche
                </Button>
              )}
            </div>
          </div>

          {/* Results */}
          {result && (
            <div className="mt-8 animate-fade-up">

                      {result.visaRequired
                        ? "Un visa est nécessaire pour entrer au Sénégal"
                        : "Vous pouvez entrer au Sénégal sans visa préalable"
                      }
                    </p>
                  </div>
                </div>

                {/* Details Grid */}
                <div className="grid grid-cols-1 md:grid-cols-2 gap-6">

                    <div className="bg-muted/50 rounded-lg p-4">
                      <p className="text-sm text-muted-foreground mb-1">Type de visa</p>
                      <p className="font-semibold text-foreground">{result.visaType}</p>
                    </div>
                  )}
                  {result.maxStayDays && (
                    <div className="bg-muted/50 rounded-lg p-4">
                      <p className="text-sm text-muted-foreground mb-1">Durée maximale de séjour</p>
                      <p className="font-semibold text-foreground">{result.maxStayDays} jours</p>
                    </div>
                  )}
                </div>

                {result.additionalDocuments && result.additionalDocuments.length > 0 && (

                    </p>
                  </div>

          )}
        </div>
      </div>
    </section>
  );
};

export default VisaSearchForm;
